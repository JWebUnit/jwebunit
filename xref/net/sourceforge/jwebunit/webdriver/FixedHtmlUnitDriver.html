19  package net.sourceforge.jwebunit.webdriver;
20  
21  import com.gargoylesoftware.htmlunit.TopLevelWindow;
22  import org.openqa.selenium.htmlunit.HtmlUnitDriver;
23  
24  public class FixedHtmlUnitDriver extends HtmlUnitDriver {
25  
26      @Override
27      public void close() {
28          if (getCurrentWindow() != null) {
29              if (getWebClient().getWebWindows().size() > 1) {
30                  ((TopLevelWindow) getCurrentWindow().getTopWindow()).close();
31              }
32              else {
33                  quit();
34              }
35          }
36      }

38  }
